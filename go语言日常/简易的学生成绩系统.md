package main

import (
	"fmt"
	"time"
)
var num int
var max int
var min int
func returnMuBan(){

	var input string
	fmt.Scanln(&input)
	fmt.Printf("正在返回查询系统\n")

	fmt.Printf("2 ")
	time.Sleep(1 * time.Second)
	fmt.Printf("1\n")

func muBan(){
	fmt.Printf("====欢迎进入查询系统====\n")
	fmt.Printf("=======1、求总分========\n")
	fmt.Printf("=======2、平均分========\n")
	fmt.Printf("=======3、最高分========\n")
	fmt.Printf("=======4、最低分========\n")
	fmt.Printf("=======5、修改分数======\n")
	fmt.Printf("=======6、查询学生成绩==\n")
	fmt.Printf("=======7、退出==========\n")
}
func z_sum(n []int){
	var sum int
	for i:=0;i<num;i++{
		sum=sum+n[i]
	}
	fmt.Printf("%d个学生的总成绩为：%d 分\n",num,sum)
	returnMuBan()
}
func z_avg(n []int){
	var sum int
	var avg float64
	for i:=0;i<num;i++{
		sum=sum+n[i]
	}
	avg=float64(sum)/float64(num)
	fmt.Printf("%d个学生的平均分为：%0.4f 分\n",num,avg)
	returnMuBan()
}
func z_max(n []int){
	max=n[0]
	for i:=0;i<num;i++{
		if n[i]>max{
			max=n[i]
		}
	}
	fmt.Printf("这%d个学生的最高分为 %d 。\n",num,max)
	returnMuBan()
}
func z_min(n []int){
	min=n[0]
	for i:=0;i<num;i++{
		if n[i]<min{
			min=n[i]
		}
	}
	fmt.Printf("这%d个学生的最低分为 %d 。\n",num,min)
	returnMuBan()
}
func z_alter(n []int){
	var zzj int
	var score int
	for i:=0;i<num;i++{
		fmt.Printf("[%d]：%d\n",i+1,n[i])
	}
	time.Sleep(1 * time.Second)
	fmt.Printf("请输入修改哪位同学的成绩\n")
	fmt.Printf("请输入编号:")
	for {
		fmt.Scanf("%d",&zzj)
		if zzj <= num {
			fmt.Printf("[%d]的成绩为：%d\n", zzj, n[zzj-1])
			time.Sleep(1 * time.Second)
			fmt.Printf("请输入你要改的成绩数值\n")
			fmt.Printf("请输入:")
			for {
				fmt.Scanf("%d",&score)
				if score>100 || score<0{
					fmt.Printf("重新输入\n")
				}else{
					break
				}
			}
			n[zzj-1] = score
			time.Sleep(1 * time.Second)
			fmt.Printf("现在[%d]的成绩为：%d\n", zzj, n[zzj-1])
			time.Sleep(1 * time.Second)
			fmt.Printf("全部成绩如下\n")
			for j := 1; j <= num; j++ {
				fmt.Printf("[%d]：%d\n", j, n[j-1])
			}
			break
		}else{
			fmt.Printf("没有此%d号的学生\n",zzj)
			fmt.Printf("重新输入\n")
		}
	}
	returnMuBan()
}

	var choose int
	fmt.Printf("现在有%d个学生,请根据编号查询---只用输入数字\n",num)
	fmt.Printf("请输入:")
ssq:	for {
	fmt.Scanf("%d",&choose)

		fmt.Printf("你输入的编号是%d\n", choose)

		break
	}else{
		fmt.Printf("没有此%d号的学生\n",choose)
		fmt.Printf("重新输入:")
	}
}
	var z int
	fmt.Printf("=======1、再次查询=======\n")
	fmt.Printf("=======2、退出系统=======\n")
	fmt.Scanf("%d",&z)
	if z==1{
		fmt.Printf("再次输入：")
		goto ssq
	}else{
		returnMuBan()
	}
}
func main(){
	var use int          //程序选择
	var choose int      //系统选择
	var Sum_Score [10000]int
	var score []int
	fmt.Printf("这是一个成绩查询系统\n")
	time.Sleep(1 * time.Second)
	fmt.Printf("现在我要开始录用成绩了\n")
	time.Sleep(1 * time.Second)
	fmt.Printf("请输入你要录用的人数\n")
	fmt.Printf("输入：")
	fmt.Scanf("%d",&num)
	fmt.Printf("开始录用%d个学生成绩\n",num)
	score =Sum_Score[ :num]
	for i:=0;i<len(score);i++{
		fmt.Printf("请输入第%d给学生的成绩\n",i+1)
		fmt.Scanf("%d",&score[i])
		if score[i]>100 || score[i]<0{
			fmt.Printf("重新输入\n")
			i--
			continue
		}
	}
	fmt.Printf("成绩录用完毕\n")
	time.Sleep(1 * time.Second)
	fmt.Printf("是否进入成绩查询系统\n")
	fmt.Printf("=======1、进入查询系统=======\n")
	fmt.Printf("=======2、退出系统===========\n")
	fmt.Printf("请选择：")
	fmt.Scanf("%d",&choose)
	fmt.Printf("请稍等\n")
	time.Sleep(1 * time.Second)
	for {
		if choose == 1 {
			muBan()
			fmt.Printf("请选择你要进入的查询\n")
			fmt.Printf("请输入:")
			fmt.Scanf("%d",&use)
			switch use {
			case 1:z_sum(score)
			case 2:z_avg(score)
			case 3:z_max(score)
			case 4:z_min(score)
			case 5:z_alter(score)
			case 6:z_select(score)
			case 7:
				fmt.Printf("请稍等\n")
				time.Sleep(1 * time.Second)
				goto zzj
			}
		} else {
			fmt.Printf("请稍等\n")
			time.Sleep(1 * time.Second)
			goto zzj
		}
	}
zzj:	fmt.Printf("==>退出成功")
